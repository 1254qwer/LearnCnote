	float sum = 0, term;
	int n = 0;

	scanf("%f", &x);
	/********************Begin********************/
	n = 1;
	int i = 1, flag = 1;
	float jiecheng = 1.00;
	do
	{
		for (i = 1; i <= 2 * n - 1; i++)

			jiecheng = jiecheng * i * 1.00;

		flag = -flag;
		jiecheng = 1.00;
		n++;
	} while (fabs(term) >= 1e-6);
	/*******************End*********************/
	printf("\nsin(%.2f) = %.2f\n", x, sum);

{
	int i;
	if (m <= n)
		i = m;
	else
		i = n;

	{
